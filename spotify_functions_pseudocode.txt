        else if string begins with song         uri_type = 'song'

        if song uri selected:
            song_uri = uri 
            audio_analysis = API function call
            audio_features = API function call
            
        else if playlist uri selected:
            playlist_uri = uri
            playlist_tracks = API call            # Gets raw playlist data

        else raise exception: not playlist or song uri
